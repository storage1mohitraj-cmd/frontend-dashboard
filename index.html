document.addEventListener("DOMContentLoaded", () => { const carousel = document.getElementById('hero-carousel'); if (!carousel) return; const items = carousel.querySelectorAll('.hero-item'); const totalItems = items.length; // Adjust radius based on screen size for better responsiveness const isMobile = window.innerWidth <= 640; const radius = isMobile ? 240 : 360; let currAngle = 0; let isDraggingCarousel = false; let startX = 0; let currentX = 0; let lastTime = 0; let velocity = 0; let autoSpin = true; let animationFrame; // Position items in 3D circle items.forEach((item, index) => { // Start Molly (index 2) at the front (angle 0) const offsetIndex = (index - 2 + totalItems) % totalItems; const itemAngle = (360 / totalItems) * offsetIndex; item.style.transform = `rotateY(${itemAngle}deg) translateZ(${radius}px)`; item.dataset.angle = itemAngle; // Handle click item.addEventListener('click', (e) => { if (Math.abs(currentX) > 10) return; // Prevent click while dragging autoSpin = false; items.forEach(i => { i.classList.remove('show-node'); }); // Toggle node if (item.classList.contains('active')) { item.classList.add('show-node'); } else { // Rotate carousel to this item const targetAngle = -itemAngle; // Calculate shortest path let diff = (targetAngle - currAngle) % 360; if (diff > 180) diff -= 360; if (diff < -180) diff += 360; currAngle += diff; updateCarousel(); item.classList.add('show-node'); } }); }); // Default active items[2].classList.add('active'); // Molly is default function updateCarousel() { carousel.style.transform = `translateZ(-${radius}px) rotateY(${currAngle}deg)`; // Update active class const snapAngle = 360 / totalItems; const normalized = Math.round(currAngle / snapAngle) * snapAngle; const activeOffsetIndex = (Math.round(-normalized / snapAngle) % totalItems + totalItems) % totalItems; const activeIndex = (activeOffsetIndex + 2) % totalItems; items.forEach((item, idx) => { if (idx === activeIndex) { item.classList.add('active'); } else { item.classList.remove('active', 'show-node'); } }); } carousel.addEventListener('mousedown', dragStart, { passive: false }); carousel.addEventListener('touchstart', dragStart, { passive: false }); document.addEventListener('mousemove', dragMove, { passive: false }); document.addEventListener('touchmove', dragMove, { passive: false }); document.addEventListener('mouseup', dragEnd); document.addEventListener('touchend', dragEnd); function dragStart(e) { if (e.target.closest('.hero-node') || e.target.closest('.hero-feed-close')) return; if (e.type === 'touchstart') e = e.touches[0]; isDraggingCarousel = true; autoSpin = false; startX = e.clientX; currentX = 0; lastTime = performance.now(); carousel.style.transition = 'none'; cancelAnimationFrame(animationFrame); } function dragMove(e) { if (!isDraggingCarousel) return; // prevent page scroll while dragging carousel on mobile if (e.type === 'touchmove') e.preventDefault(); if (e.type === 'touchmove') e = e.touches[0]; const now = performance.now(); const dt = now - lastTime; const deltaX = e.clientX - startX; currentX += deltaX; startX = e.clientX; if (dt > 0) { velocity = deltaX / dt; } lastTime = now; currAngle += (deltaX * 0.4); updateCarousel(); } function dragEnd(e) { if (!isDraggingCarousel) return; isDraggingCarousel = false; carousel.style.transition = 'transform 0.6s cubic-bezier(0.25, 1, 0.5, 1)'; // Add momentum currAngle += velocity * 100; // Snap to nearest const snapAngle = 360 / totalItems; currAngle = Math.round(currAngle / snapAngle) * snapAngle; updateCarousel(); velocity = 0; // Resume auto spin after 5 seconds of inactivity setTimeout(() => { if (!isDraggingCarousel && !Array.from(items).some(i => i.classList.contains('show-node'))) { autoSpin = true; } }, 5000); } // Auto spin loop function loop() { if (autoSpin) { currAngle -= 0.12; carousel.style.transition = 'none'; updateCarousel(); } animationFrame = requestAnimationFrame(loop); } loop(); // Initial update updateCarousel(); });
